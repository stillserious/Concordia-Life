  <script>
    const partNames = {
      body: "Car Body",
      hood: "Hood",
      trunk: "Trunk",
      'left-door': "Left Door",
      'right-door': "Right Door",
      'front-wheel': "Front Wheel",
      'rear-wheel': "Rear Wheel",
      roof: "Roof",
      windshield: "Windshield",
      'rear-window': "Rear Window"
    };

    document.querySelectorAll('.car-part').forEach(part => {
      part.addEventListener('click', function() {
        // Remove highlight from all
        document.querySelectorAll('.car-part').forEach(el => el.classList.remove('highlight'));
        // Highlight clicked part
        this.classList.add('highlight');
        // Show info
        document.getElementById('info').textContent = 'Selected: ' + (partNames[this.id] || this.id);
      });
    });
  </script>
